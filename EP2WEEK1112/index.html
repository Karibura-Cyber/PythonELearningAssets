Part 1

สร้างหน้าต่างเกม (Game Loop)

ทุกเกม pygame เริ่มด้วย Game Loop — วนซ้ำ 60 ครั้งต่อวินาที: รับ Event → Update → วาดภาพ

1

pygame.init() + สร้างหน้าต่าง

screen = pygame.display.set_mode((800, 600))

2

while running: (Game Loop)

วนซ้ำตลอดจนกว่าจะปิดเกม — ภายในมี 3 ส่วน: Event, Update, Draw

3

screen.fill((0,0,0)) — เติมพื้นหลัง

ต้องเติมสีทับก่อนทุก frame ไม่งั้นรูปเก่าจะค้างอยู่

4

pygame.display.flip() + clock.tick(60)

แสดงผลบนจอ + จำกัด FPS ที่ 60

import pygame, random pygame.init() WIDTH, HEIGHT = 800, 600 screen = pygame.display.set_mode((WIDTH, HEIGHT)) pygame.display.set_caption("Space Shooter - by Kru Joke") clock = pygame.time.Clock() running = True while running: screen.fill((0, 0, 0)) # เติมพื้นหลังสีดำ for event in pygame.event.get(): if event.type == pygame.QUIT: running = False pygame.display.flip() clock.tick(60) pygame.quit()
Part 2

จรวด และการควบคุม 🚀

โหลด rocket.png กำหนดตำแหน่ง และรับ input จากแป้นพิมพ์ พร้อมป้องกันวิ่งทะลุขอบจอ

# --- พิมพ์ไว้ด้านบน (ก่อน running = True) --- rocket_img = pygame.image.load("rocket.png") rocket_img = pygame.transform.scale(rocket_img, (60, 80)) player_x, player_y = 370, 500 player_change_x = 0 player_speed = 5 # --- ใน for event --- if event.type == pygame.KEYDOWN: if event.key == pygame.K_LEFT: player_change_x = -player_speed if event.key == pygame.K_RIGHT: player_change_x = player_speed if event.type == pygame.KEYUP: if event.key in (pygame.K_LEFT, pygame.K_RIGHT): player_change_x = 0 # --- ใน while loop (ก่อนวาดรูป) --- player_x += player_change_x if player_x < 0: player_x = 0 elif player_x > WIDTH - 60: player_x = WIDTH - 60 screen.blit(rocket_img, (player_x, player_y))
Part 3

อุกกาบาต และการตก ☄️

อุกกาบาตเพิ่มค่า Y ทุก frame (ตกลงมา) ถ้าหลุดพ้นขอบล่าง สุ่มเกิดใหม่ด้านบน

# --- ด้านบน --- meteor_img = pygame.image.load("meteorite.png") meteor_img = pygame.transform.scale(meteor_img, (50, 50)) enemy_x = random.randint(0, WIDTH - 50) enemy_y = -50 enemy_speed = 4 # --- ใน while loop --- enemy_y += enemy_speed # เคลื่อนลง if enemy_y > HEIGHT: # ถ้าตกพ้นขอบ enemy_y = -50 enemy_x = random.randint(0, WIDTH - 50) screen.blit(meteor_img, (enemy_x, enemy_y))
Part 4

คะแนน และระบบอาวุธ ⚡

ใช้ List เก็บกระสุนหลายนัด กดยิงแล้ว append ใส่ List วนลูปขยับและตรวจ collision

# --- ด้านบน --- lasers = [] laser_speed = 10 score = 0 score_font = pygame.font.Font(None, 36) # --- กด SPACE ยิง --- if event.key == pygame.K_SPACE: lasers.append([player_x + 27, player_y]) # --- ขยับกระสุน + ลบถ้าพ้นจอ --- for laser in lasers[:]: laser[1] -= laser_speed if laser[1] < 0: lasers.remove(laser) # --- ตรวจ Collision กระสุน vs อุกกาบาต --- enemy_box = pygame.Rect(enemy_x, enemy_y, 50, 50) for laser in lasers[:]: laser_box = pygame.Rect(laser[0], laser[1], 5, 20) if laser_box.colliderect(enemy_box): lasers.remove(laser); score += 10 enemy_y = -50; enemy_x = random.randint(0, WIDTH - 50) break # --- แสดงคะแนน --- score_text = score_font.render("Score: " + str(score), True, (255,255,255)) screen.blit(score_text, (10, 10))
Part 5

ชีวิต และ Game Over ❤️

ตรวจ collision ระหว่างยานกับอุกกาบาต ถ้าชนให้ชีวิตลด ชีวิตหมดแสดงหน้าจอ Game Over

# --- ด้านบน --- lives = 3 game_over = False my_font = pygame.font.Font(None, 60) # --- ใน while: ตรวจยานชนอุกกาบาต --- player_box = pygame.Rect(player_x, player_y, 60, 80) if player_box.colliderect(enemy_box): lives -= 1 enemy_y = -50 enemy_x = random.randint(0, WIDTH - 50) if lives <= 0: game_over = True # --- ครอบระบบทั้งหมดด้วย if not game_over --- if not game_over: # [โค้ดอัปเดต + วาดทั้งหมดอยู่ที่นี่] else: screen.fill((150, 0, 0)) go_text = my_font.render("GAME OVER", True, (255,255,255)) screen.blit(go_text, go_text.get_rect(center=(WIDTH//2, HEIGHT//2)))
